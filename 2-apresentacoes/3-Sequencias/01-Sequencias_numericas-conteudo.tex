\section{Definição e Convergência}

%------------------------------------------------------------------------------%
\begin{frame}{Definição de Sequências}
    Uma sequência é uma \emph{lista de números} em uma ordem determinada
    \[a_1,\;a_2,\;a_3,\; \dots,\; a_n,\; \dots\]

    \pause
    $\memph{a_n}$ são os \emph{termos} e $\memph{n}$ são os \emph{índices} da sequência

    \pause
    \vspace{\baselineskip}
    Uma sequência também é uma função
    \[f:\N\to\R \qquad \memph{a_n = f(n)}\]
\end{frame}

\begin{frame}{Convergência de Sequências}
    $(a_n)$ \emph{converge} para $L$ se, para todo $\memph{\varepsilon}>0$
    existe $\memph{N}\in \N$ tal que
    \[
    	n>\memph{N} \quad \Rightarrow  \quad \abs{a_n-L}<\memph{\varepsilon}
    \]

    \pause
    \vspace{\baselineskip}
    Notação
    \[\lim_{n\to \infty} a_n=L \qquad\text{ ou }\qquad a_n\to L \qquad\text{ ou }\qquad \lim a_n=L\]

    \pause
    \vspace{\baselineskip}
    Uma sequência convergente $(a_n)$ tem apenas \emph{um limite}.
\end{frame}

\begin{frame}{Explorando a Definição}
    Percebemos que a sequência \(a_n = \frac{1}{n}\) converge para $L = 0$

    \pause
    \vspace{\baselineskip}
    Para $\varepsilon=1$     \tabto{28mm} basta      \tabto{50mm} $N=1$    \\
    Para $\varepsilon=0,001$ \tabto{28mm} precisamos \tabto{50mm} $N=1000$

    \pause
    \vspace{\baselineskip}
    Quantos passos precisamos que garantir uma precisão de $\varepsilon$?

\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Explorando a Definição}
    Precisamos encontrar um $N$ que garanta
    \(\abs{a_n-L}<\varepsilon\)
    para todo $n>N$
    \pause
    \[\abs{\,a_n-L\,} = \abs{\,\frac{1}{n}-0\,} = \frac{1}{\,n\,} \]
    \pause
    então
    \[
        \abs{\,a_n-L\,}<\varepsilon
        \pause
        \quad\Leftrightarrow\quad
        \frac{1}{\,n\,} < \varepsilon
        \pause
        \quad\Leftrightarrow\quad
        \frac{1}{\,\varepsilon\,} < n
    \]
    \pause
    escolhemos
    \[
        N = \ceil*{\;\frac{1}{\;\varepsilon\;}\;}
        \qquad\qquad\text{menor inteiro maior do que } \frac{1}{\;\varepsilon\;}
    \]

\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Divergência ao infinito}
    $(a_n)$ \emph{\textbf{diverge} ao infinito} se dado $\memph{M}\in\R$ existe $\memph{N}\in\N$ tal que
    \[ n>\memph{N} \quad \Rightarrow \quad a_n > \memph{M} \]

    \pause
    \vspace{\baselineskip}
    Notação
    \[ \lim_{n\to\infty} a_n=\infty
    \qquad\text{ ou }\qquad
    a_n\to \infty
    \qquad\text{ ou }\qquad
    \lim a_n=\infty
    \]

\end{frame}

\begin{frame}{Propriedades de Sequências Convergentes}
    Sejam $(a_n)$ e $(b_n)$ sequências \emph{convergentes}, então temos as regras: \\[4mm]
    \pause
    \begin{enumerate}[<+->]
        \item $\lim (a_n+b_n) = \lim a_n + \lim b_n$                                                 \tabto{77mm} Soma                      \\[5mm]
        \item $\lim (a_n-b_n) = \lim a_n - \lim b_n$                                                 \tabto{77mm} Diferença                 \\[5mm]
        \item $\lim (k a_n)   = k\lim a_n,\quad \forall\; k\in\R$                                    \tabto{77mm} Multiplicação por escalar \\[5mm]
        \item $\lim (a_n\,b_n)= (\lim a_n)(\lim b_n)$                                                \tabto{77mm} Produto                   \\[5mm]
        \item \(\lim \frac{a_n}{b_n} = \frac{\lim a_n}{\lim b_n}\quad\) desde que $\;\lim b_n \neq 0$  \tabto{77mm} Quociente
    \end{enumerate}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Propriedades de Sequências Convergentes}
	Notem que o contrário não vale

	\pause
	\vspace{\baselineskip}
	Considerem as sequências divergentes
	\[
		a_n = n \qquad\qquad b_n = -n
	\]
	sua soma é convergente
	\[
		c_n \;=\; a_n + b_n \;=\; n + (-n) \;=\; 0
	\]

\end{frame}

\section{Teoremas}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema do Confronto}
  Sejam $(a_n)$, $(b_n)$ e $(c_n)$ tais que
  \[
   	\memph{a_n \;\leq\; b_n \;\leq\; c_n} \qquad\qquad \forall\; n > N
 	\]

	\pause
  \vspace{\baselineskip}
  Se $\quad a_n\to L\quad$ e $\quad c_n\to L\quad$ então $\quad b_n\to L$
\end{frame}

\begin{frame}{Teorema da Função Contínua}
É verdade que \quad
\emph{\(
  \lim_{n\to \infty} f(a_n) = f\!\left(\lim_{n\to \infty} a_n\right)
\)} \quad?

\pause
\vspace{2\baselineskip}
Se $a_n$ é uma sequência numérica real convergente, $a_n\to L$, e
$f\colon D\in\R\to\R$ é uma função real contínua em $D$,
com $L\in D$
então
\[
  \lim f(a_n) = f(L)
\]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Limite de sequência definida por função}
    Se $f$ é uma função e $\quad a_n=f(n)\quad$ então
    \[
    	\lim_{x\to\infty}f(x)=L \quad \Rightarrow \quad \lim_{n\to \infty} a_n=L
    \]

    \pause
    \vspace{\baselineskip}
    Esse resultado nos permite usar a regra de l'Hôpital (em $x$)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Sequências Limitadas}
	Uma sequência $(a_n)$ é

	\vspace{\baselineskip}
	\emph{Limitada inferiormente} \tabto{50mm} se existe $m\in\R$ tal que \emph{$m\leq a_n$} para todo $n\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada superiormente} \tabto{50mm} se existe $M\in\R$ tal que \emph{$M\geq a_n$} para todo $n\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada}               \tabto{50mm} se é limitada superiormente e inferiormente

\end{frame}

\begin{frame}{Sequências Crescentes e Decrescentes}
	Uma sequência $(a_n)$ é

	\vspace{\baselineskip}
	\emph{Crescente}   \tabto{30mm} se \emph{$\;a_n \leq a_{n+1}\;$} para todo $n\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Decrescente} \tabto{30mm} se \emph{$\;a_n \geq a_{n+1}\;$} para todo $n\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Monótona}    \tabto{30mm} se é crescente ou decrescente
\end{frame}

\begin{frame}{Convergência de sequência monótona}
	Se uma sequência $(a_n)$ é \emph{limitada e monotônica}, então ela é \emph{convergente}

	\pause
	\vspace{2\baselineskip}
	Importante porque permite decidir a convergência sem calcular o limite
\end{frame}
