\begin{frame}{Caso 4}
  $Q(x)$ contém fatores quadráticos irredutíveis repetidos
  \[
    Q(x) = \cdots  \left( ax^2 + bx + c \right)^r
    \qquad
    \text{ com } b^2-4ac<0
  \]

  \pause
  Expansão
  \[
    \frac{P(x)}{Q(x)}
    = \cdots
    + \frac{A_1x+B_1}{ax^2+bx+c}
    + \frac{A_2x+B_2}{(ax^2+bx+c)^2}
    + \cdots
    + \frac{A_rx+B_r}{(ax^2+bx+c)^r}
  \]
\end{frame}

\begin{frame}{Exemplo}
\[
  \frac{2x}{(x-9)(x^2+4)^2}
  = \frac{A}{x-9}
  + \frac{Bx+C}{x^2+4}
  + \frac{Dx+E}{(x^2+4)^2}
\]
\end{frame}
