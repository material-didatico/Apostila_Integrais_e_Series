\section{Função Definida por Série de Potências}

%------------------------------------------------------------------------------%
\begin{frame}{Função Definida por Série de Potências}
  Se
  \[
    \sum_{n=0}^\infty c_n (x-a)^n
  \]
  \pause
  possui um raio de convergência \(R>0\),
  \pause
  então a função
  \[
    f(x)=\sum_{n=0}^\infty c_n(x-a)^n
  \]
  \pause
  está definida no intervalo $(a-R,\,a+R)$ \\[3mm] \pause
  é infinitamente derivável nesse intervalo
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Manipulando Funções Definidas por Séries}
  Podemos fazer mudanças de variáveis e somar as séries

  \pause
  \vspace{2\baselineskip}
  Desde que estejamos atentos ao intervalo de convergência
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Escreva \(f(x) = \frac{1}{1-x^2}\) como uma série de potências

  \pause
  \vspace{\baselineskip}
  Sabemos que
  \[
    \frac{1}{1-u}
    \;=\; \sum_{n=0}^\infty u^n
    \;=\; 1 + u + u^2 + u^3 + \cdots + u^n + \cdots
  \]
  para \(\abs{u}<1\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \onslide<+->{Fazendo \(u = x^2\)}\onslide<+->{, temos
  \[
    \frac{1}{1-u}
    \;=\; \sum_{n=0}^\infty u^n
    \;=\; 1 + u + u^2 + u^3 + \cdots + u^n + \cdots
  \]}
  \[
    \onslide<+->{\frac{1}{1-x^2}
    \;=\; \sum_{n=0}^\infty (x^2)^n}
    \onslide<+->{\;=\; \sum_{n=0}^\infty x^{2n}}
    \onslide<+->{\;=\; 1 + x^2 + x^4 + x^6 + \cdots + x^{2n} + \cdots}
  \]

  \onslide<+->{Analisando o intervalo de convergência}
  \begin{align*}
    \onslide<+->{\abs{u} & < 1        \\}
    \onslide<+->{x^2     & < 1        \\}
    \onslide<+->{\abs{x} & < \sqrt{1} \\}
    \onslide<+->{-1 < x  & < 1}
  \end{align*}
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Escreva \(f(x) = \frac{1}{x+2}\) como uma série de potências

  \pause
  \vspace{\baselineskip}
  Sabemos que
  \[
    \frac{1}{1-u}
    \;=\; \sum_{n=0}^\infty u^n
    \;=\; 1 + u + u^2 + u^3 + \cdots + u^n + \cdots
  \]
  para \(\abs{u}<1\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Precisamos manipular $f$
  \pause
  \[
    f(x)
    \;=\; \frac{1}{x+2}
    \pause
    \;=\; \frac{1}{2\left(1+\dfrac{x}{2}\right)}
    \pause
    \;=\; \frac{1}{2}\,\frac{1}{\left[1-\left(-\dfrac{x}{2}\right)\right]}
  \]

  \pause
  Fazendo \(u=-\frac{x}{2}\)\pause, temos
  \[
    f(x)
    \;=\; \frac{1}{2} \, \sum_{n=0}^\infty \left(-\frac{x}{2}\right)^n
    \pause
    \;=\; \sum_{n=0}^\infty \frac{1}{2} \, \frac{(-1)^n\,x^n}{2^n}
    \pause
    \;=\; \sum_{n=0}^\infty \frac{(-1)^n}{2^{n+1}}\,x^n
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \onslide<+->{Analisando o intervalo de convergência}
  \begin{align*}
    \onslide<+->{\abs{u}    \;\;        & < 1 \\[1mm]}
    \onslide<+->{\abs{-\frac{x}{2}}     & < 1 \\[1mm]}
    \onslide<+->{\frac{\abs{x}}{2} \;\; & < 1 \\[1mm]}
    \onslide<+->{\abs{x} \;\;           & < 2}
  \end{align*}
  \onslide<+->{A série converge para \(x\in(-2, 2)\)}
\end{frame}

\section{Derivada da Série}

%------------------------------------------------------------------------------%
\begin{frame}{Derivação Termo a Termo}
  Podemos calcular a derivada de $f(x)$ construindo a série
  \begin{align*}
    \action<+->{f'(x) & = {\color{blue}\frac{d}{dx} \Bigg(} \sum_{n=0}^\infty c_n(x-a)^n {\color{blue}\Bigg)} \\[2mm]}
    \action<+->{      & = \sum_{n=0}^\infty {\color{blue}\frac{d}{dx} \Big(} c_n(x-a)^n {\color{blue}\Big)} \\[2mm]}
    \action<+->{      & = \sum_{{\color<5->{red}n=1}}^\infty c_n \,n\, \big(x-a\big)^{n-1}}
    \action<5->{\qquad\text{Note que o índice dessa série começa em {\color{red}1}}}
  \end{align*}
  \action<+->{que converge em $(a-R, a+R)$}
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcule a derivada da função de Bessel
  \pause
  \[
  \bessel_0(x) = \sum_{n=0}^\infty \frac{\,(-1)^n\,x^{2n}\,}{2^{2n}\,(n!)^2}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \begin{columns}
    \column{0.5\baselineskip}
    \begin{align*}
      \onslide<+->{\bessel'_0(x)}
      \onslide<+->{& = \frac{d}{dx} \left(\sum_{n=0}^\infty \frac{(-1)^nx^{2n}}{2^{2n}(n!)^2}\right) \\[3mm]}
      \onslide<+->{& = \sum_{n=0}^\infty \frac{d}{dx} \left(\frac{(-1)^nx^{2n}}{2^{2n}(n!)^2}\right) \\[3mm]}
      \onslide<+->{& = \sum_{n=0}^\infty \frac{(-1)^n}{2^{2n}(n!)^2} \frac{d}{dx} \left(x^{2n}\right)}
    \end{align*}
    \column{0.5\baselineskip}
    \begin{align*}
      \onslide<+->{\bessel'_0(x)
        & = \sum_{n=1}^\infty \frac{(-1)^n}{2^{2n}(n!)^2} \; 2n x^{2n-1} \\[3mm]}
      \onslide<+->{& = \sum_{n=1}^\infty \frac{\,(-1)^n\, n\, x^{2n-1}\,}{(n!)^2\,2^{2n-1}}}
    \end{align*}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Qual a inclinação da reta tangente ao gráfico da função de Bessel em $x=2$?
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Já verificamos que
  \[
    \bessel'_0(x)
    = \sum_{n=1}^\infty \frac{\,(-1)^n\, n\, x^{2n-1}\,}{2^{2n-1}(n!)^2}
  \]

  \pause
  Avaliando a derivada em $x=2$, temos
  \[
    \bessel'_0(2)
    = \sum_{n=1}^\infty \frac{\,(-1)^n\, n\, x^{2n-1}\,}{2^{2n-1}(n!)^2} \evalat{x=2}{}
    \pause
    = \sum_{n=1}^\infty \frac{\,(-1)^n\, n\,}{(n!)^2} \frac{\,2^{2n-1}\,}{2^{2n-1}}
    \pause
    = \sum_{n=1}^\infty \frac{\,(-1)^n\, n\,}{(n!)^2}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Determine a inclinação da reta tangente ao gráfico da função
  \[
      f(x) = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}
      \qquad\qquad
      \abs{x}<1
  \]
  em \(x=\frac{1}{4}\)
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  \onslide<+->{Calculando a derivada de $f$, temos}
  \begin{align*}
    \onslide<+->{f'(x)}
    \onslide<+->{& = \frac{d}{dx}\left(\sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}\right) \\[2mm]}
    \onslide<+->{& = \sum_{n=0}^{\infty} \frac{d}{dx}\left(\frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}\right) \\[2mm]}
    \onslide<+->{& = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} \frac{d}{dx}\left(x^{2n+1}\right) \\[2mm]}
    \onslide<+->{& = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} (2n+1)x^{2n}}
    \onslide<+->{\;=\; \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2} x^{2n}}
  \end{align*}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \[
    f'\left(\sfrac{1}{4}\right)
    \pause
    = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2} x^{2n} \evalat{x=\sfrac{1}{4}}{}
    \pause
    = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2} \frac{1}{4^{2n}}
    \pause
    = \sum_{n=0}^{\infty} \frac{(2n)!}{4^{3n} (n!)^2}
  \]
\end{frame}

\section{Integral da Série}

%------------------------------------------------------------------------------%
\begin{frame}{Integração Termo a Termo}
  Para calcularmos a integral de $f$ construímos a série
  \begin{align*}
    \action<+->{\int f(x)dx & = {\color{blue}\int} \left(\sum_{n=0}^\infty c_n(x-a)^n\right){\color{blue}dx} \\[2mm]}
    \action<+->{            & = \sum_{n=0}^\infty \left({\color{blue}\int} c_n(x-a)^n {\color{blue}dx}\right) \\[2mm]}
    \action<+->{            & = \sum_{n=0}^\infty c_n\frac{(x-a)^{n+1}}{n+1}+ C}
\end{align*}
  \action<+->{que também converge em $(a-R, a+R)$}
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcule a integral da função de Bessel
  \pause
  \[
    \bessel_0(x) = \sum_{n=0}^\infty \frac{\,(-1)^n\,x^{2n}\,}{2^{2n}\,(n!)^2}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \begin{align*}
    \onslide<+->{\int \bessel_0(x) dx}
    \onslide<+->{& = \int \left(\sum_{n=0}^\infty \frac{\,(-1)^n\,x^{2n}\,}{2^{2n}\,(n!)^2}\right) dx \\[3mm]}
    \onslide<+->{& = \sum_{n=0}^\infty \left(\int \frac{\,(-1)^n\,x^{2n}\,}{2^{2n}\,(n!)^2} dx\right) \\[3mm]}
    \onslide<+->{& = \sum_{n=0}^\infty \frac{\,(-1)^n\,}{2^{2n}\,(n!)^2} \int x^{2n} dx \\[3mm]}
    \onslide<+->{& = C + \sum_{n=0}^\infty \frac{\,(-1)^n\,}{2^{2n}\,(n!)^2}\, \frac{x^{2n+1}}{2n+1}}
    \onslide<+->{\;=\; C + \sum_{n=0}^\infty \frac{\,(-1)^n\,x^{2n+1}\,}{2^{2n}(2n+1)\,(n!)^2}}
  \end{align*}
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcule a integral da função
  \[
    f(x) = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}
    \qquad\qquad
    \abs{x}<1
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  \begin{align*}
    \onslide<+->{\int f(x) dx}
    \onslide<+->{& = \int \left(\sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}\right) dx \\[2mm]}
    \onslide<+->{& = \sum_{n=0}^{\infty} \int \left(\frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}\right) dx \\[2mm]}
    \onslide<+->{& = \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} \int x^{2n+1} dx \\[2mm]}
    \onslide<+->{& = C + \sum_{n=0}^{\infty} \frac{(2n)!}{4^n (n!)^2 (2n+1)} \frac{x^{2n+2}}{2n+2}}
    \onslide<+->{\;=\; C + \sum_{n=0}^{\infty} \frac{(2n)!\,x^{2n+2}}{4^n (n!)^2 (2n+2)(2n+1)}}
  \end{align*}
\end{frame}
