\section{Teorema do Confronto}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema do Confronto}
  Sejam $(a_k)$, $(b_k)$ e $(c_k)$ tais que
  \[
   	\memph{a_k \;\leq\; b_k \;\leq\; c_k} \qquad\qquad \forall\; k > N
 	\]

	\pause
  \vspace{\baselineskip}
  Se $\quad a_k\to L\quad$ e $\quad c_k\to L\quad$ entûÈo $\quad b_k\to L$
\end{frame}

\begin{frame}{Exemplo}
  Calcular \(\lim\left[\cos\left(\alpha^k\right)\right]\) com $0<\alpha<1$

  \pause
  \vspace{\baselineskip}
  \qquad Considerando a sequûˆncia \( a_k = \alpha^k\)

  \pause
  \qquad Sabemos que \(\lim a_k=0\)

  \pause
  \qquad A funûÏûÈo cosseno ûˋ contûÙnua em zero

  \pause
  \vspace{0.5\baselineskip}
  Portanto
  \[
    \lim\cos(\alpha^k)
    = \lim\cos(a_k)
    \pause
    = \cos\big(\lim a_k\big)
    \pause
    = \cos(0)
    \pause
    = 1
  \]
\end{frame}

\section{Sequûˆncia Definida por FunûÏûÈo}

%------------------------------------------------------------------------------%
\begin{frame}{Limite de sequûˆncia definida por funûÏûÈo}
    Se $f$ ûˋ uma funûÏûÈo real e \(a_k=f(k)\) entûÈo

    \[
    	\lim_{x\to\infty} f(x) = L \quad \Rightarrow \quad
      \lim_{k\to\infty} a_k = L
    \]

    \pause
    \vspace{\baselineskip}
    Esse resultado nos permite usar a regra de l'HûÇpital (em $x$)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  Mostre que \( \lim\frac{5k^4}{k^4+8k^3} = 5 \)

  \pause
  A funûÏûÈo
  \[
    f(x) = \frac{5x^4}{x^4+8x^3}
    \hspace{50mm}
    \emph{x\in\R}
  \]
  estûÀ definida para todo $x>0$
  \pause
  e a sequûˆncia
  \[
    a_k = f(k)=\frac{5k^4}{k^4+8k^3}
    \hspace{40.5mm}
    \emph{k\in\N}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  \onslide<+->{Podemos aplicar a regra de L'HûÇpital \emph{em $x$},}
  \onslide<+->{pois temos uma indeterminaûÏûÈo $\sfrac{0}{0}$}
  \begin{align*}
    \onslide<+->{    \lim_{k\to\infty} \frac{5k^4}{k^4+8k^3}}
    \onslide<+->{  = \lim_{x\to\infty} \frac{5x^4}{x^4+8x^3}}
    \onslide<+->{& = \lim_{x\to\infty} \frac{20x^3}{4x^3+24x^2} \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{60x^2}{12x^2+48x}  \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{120x}{24x+48}      \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{120}{24}           \\[2mm]}
    \onslide<+->{& = 5}
  \end{align*}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  \onslide<+->{Mûˋtodo mais apropriado}
  \begin{align*}
    \onslide<+->{    \lim_{k\to\infty} \frac{5k^4}{k^4+8k^3}}
    \onslide<+->{& = \lim_{k\to\infty} \frac{\dfrac{\,5k^4\,}{k^4}}{\;\dfrac{\;k^4+8k^3\;}{k^4}\;} \\[3mm]}
    \onslide<+->{& = \lim_{k\to\infty} \frac{5}{1+\dfrac{8}{k}} \\[2mm]}
    \onslide<+->{& = \frac{5}{\displaystyle 1 + 8\lim_{k\to\infty} \frac{1}{k}} \\[2mm]}
    \onslide<+->{& = 5}
  \end{align*}
\end{frame}

\section{Sequûˆncia Limitadas}

%------------------------------------------------------------------------------%
\begin{frame}{Sequûˆncias Limitadas}
	Uma sequûˆncia $(a_k)$ ûˋ

	\pause
  \vspace{\baselineskip}
	\emph{Limitada inferiormente} \tabto{50mm} se existe $m\in\R$ tal que \emph{$m\leq a_k$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada superiormente} \tabto{50mm} se existe $M\in\R$ tal que \emph{$M\geq a_k$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada}               \tabto{50mm} se ûˋ limitada superiormente e inferiormente

\end{frame}

\begin{frame}{Sequûˆncias Crescentes e Decrescentes}
	Uma sequûˆncia $(a_k)$ ûˋ

	\vspace{\baselineskip}
	\emph{Crescente}   \tabto{30mm} se \emph{$\;a_k \leq a_{k+1}\;$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Decrescente} \tabto{30mm} se \emph{$\;a_k \geq a_{k+1}\;$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Monû°tona}    \tabto{30mm} se ûˋ crescente ou decrescente
\end{frame}
