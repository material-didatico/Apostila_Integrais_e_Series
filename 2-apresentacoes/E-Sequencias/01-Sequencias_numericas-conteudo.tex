\section{Definição}

%------------------------------------------------------------------------------%
\begin{frame}{Definição de Sequências}
    Uma sequência é uma \emph{lista de números} em uma ordem determinada
    \[a_1,\;a_2,\;a_3,\; \dots,\; a_k,\; \dots\]

    \pause
    $\memph{a_k}$ são os \emph{termos}
    \pause
    \;e\; $\memph{k}$ são os \emph{índices} da sequência

    \pause
    \vspace{\baselineskip}
    Uma sequência também é uma função
    \[f\colon\N\to\R \qquad \memph{a_k = f(k)}\]
\end{frame}

\section{Convergência}

%------------------------------------------------------------------------------%
\begin{frame}{Convergência de Sequências}
    $(a_k)$ \emph{converge} para $L$ se, para todo $\memph{\varepsilon}>0$
    existe $\memph{N}\in \N$ tal que
    \[
    	k>\memph{N} \quad \Rightarrow  \quad \abs{a_k-L}<\memph{\varepsilon}
    \]

    \pause
    \vspace{\baselineskip}
    Notação
    \[
      \lim_{k\to \infty} a_k = L \qquad\text{ ou }\qquad
      a_k \to L                  \qquad\text{ ou }\qquad
      \lim a_k = L
    \]

    \pause
    \vspace{\baselineskip}
    Uma sequência convergente $(a_k)$ tem apenas \emph{um limite}.
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Explorando a Definição}
    Percebemos que a sequência \(a_k = \frac{1}{k}\) converge para $L = 0$

    \pause
    \vspace{\baselineskip}
    Para $\varepsilon=1$     \tabto{28mm} basta      \tabto{50mm} $N=1$

    \pause
    Para $\varepsilon=0,001$ \tabto{28mm} precisamos \tabto{50mm} $N=1000$

    \pause
    \vspace{\baselineskip}
    Quantos passos precisamos para garantir uma precisão de $\varepsilon$ qualquer?

\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Explorando a Definição}
    Precisamos encontrar um $N$ que garanta
    \(\abs{a_k-L}<\varepsilon\)
    para todo $k>N$
    \pause
    \[\abs{\,a_k-L\,} = \abs{\,\frac{1}{k}-0\,} = \frac{1}{\,k\,} \]
    \pause
    então
    \[
        \abs{\,a_k-L\,}<\varepsilon
        \pause
        \quad\Leftrightarrow\quad
        \frac{1}{\,k\,} < \varepsilon
        \pause
        \quad\Leftrightarrow\quad
        \frac{1}{\,\varepsilon\,} < k
    \]
    \pause
    escolhemos
    \[
        N = \ceil{\;\frac{1}{\;\varepsilon\;}\;}
        \qquad\qquad\text{menor inteiro maior ou igual a } \frac{1}{\;\varepsilon\;}
    \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Divergência ao infinito}
    $(a_k)$ \emph{\textbf{diverge} ao infinito} se dado
    $\memph{M}\in\R$ existe $\memph{N}\in\N$ tal que
    \[
      k > \memph{N} \quad \Rightarrow \quad a_k > \memph{M}
    \]

    \pause
    \vspace{\baselineskip}
    Notação
    \[ \lim_{k\to\infty} a_k=\infty
    \qquad\text{ ou }\qquad
    a_k\to \infty
    \qquad\text{ ou }\qquad
    \lim a_k=\infty
    \]

\end{frame}

\begin{frame}{Casos Particulares Importantes}
  \begin{itemize}[<+->]
    \setlength\itemsep{1.5em}
    \item Se $\abs{r}<1$, \tabto{25mm} então \(\lim r^k=0\)
    \item Se $r = 1$,     \tabto{25mm} então \(\lim r^k=1\)
    \item Se $r > 1$,     \tabto{25mm} então \(\lim r^k=\infty\)
    \item Se $r \leq -1$, \tabto{25mm} então \(\lim r^k\) diverge
  \end{itemize}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Propriedades de Sequências Convergentes}
    Sejam $(a_k)$ e $(b_k)$ sequências \emph{convergentes}, então temos as regras \\[4mm]
    \pause
    \begin{enumerate}[<+->]
        \item $\lim (a_k+b_k) = \lim a_k + \lim b_k$                                                 \tabto{79mm} Soma                      \\[5mm]
        \item $\lim (a_k-b_k) = \lim a_k - \lim b_k$
        \tabto{79mm} Diferença                 \\[5mm]
        \item $\lim (c a_k)   = c\lim a_k,\quad \forall\; c\in\R$                                    \tabto{79mm} Multiplicação por escalar \\[5mm]
        \item $\lim (a_k\,b_k)= (\lim a_k)(\lim b_k)$                                                \tabto{79mm} Produto                   \\[5mm]
        \item \(\lim \frac{a_k}{b_k} = \frac{\lim a_k}{\lim b_k}\quad\) desde que $\;\lim b_k \neq 0$  \tabto{79mm} Quociente
    \end{enumerate}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Propriedades de Sequências Convergentes}
	Note que o contrário não vale

	\pause
	\vspace{\baselineskip}
	Considere as sequências divergentes
	\[
		a_k = k \qquad\qquad b_k = -k
	\]
	\pause
  a soma é convergente
	\[
		c_k \;=\; a_k + b_k \;=\; k + (-k) \;=\; 0
	\]
\end{frame}

\begin{frame}{Exemplo}
  \onslide<+->{
  Sendo $0<r<1$, calcule o limite da sequência
  \(
    a_k = \dfrac{\;-\dfrac{1}{r^k}+7\;}{ 1 + \dfrac{5}{r^k}}
  \)}
  \begin{align*}
    \onslide<+->{\lim a_k
                 & = \lim \left(\frac{-\dfrac{1}{r^k}+7}{1+ \dfrac{5}{r^k}}\right) }
    \onslide<+->{  = \lim \left(\frac{\dfrac{-1+7r^k}{r^k}}
                                     {\dfrac{r^k + 5}{r^k}}\right) }
    \onslide<+->{  = \lim \left(\frac{-1+7r^k}{r^k+5}\right) \\[3mm]}
    \onslide<+->{& = \dfrac{-1+7\lim(r^k)}{\lim(r^k) + 5}}
    \onslide<+->{\hspace{70mm} \lim r^k=0\\[3mm]}
    \onslide<+->{& = -\frac{1}{5}}
  \end{align*}
\end{frame}
