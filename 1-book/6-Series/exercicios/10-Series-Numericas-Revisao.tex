\begin{exercise}
  Considere as sequûˆncias
  \[
    a_n = \ln\left(1+\frac{1}{n^2}\right)
    \qquad\text{e}\qquad
    b_n = \frac{1}{n^2}
  \]
  \vspace{-\baselineskip}
  \begin{alphalist}
    \item Quais das seguintes sûˋries convergem e quais divergem?
          No item (i) use o teste da integral diretamente, nûÈo use que ûˋ uma sûˋrie~$p$.
    \begin{romanlist}[2]
      \item \( \sum_{n=1}^\infty b_n \)
      \item \( \sum_{n=1}^\infty a_n \)
      \item \( \sum_{n=1}^\infty \frac{a_n}{b_n} \)
      \item \( \sum_{n=1}^\infty a_n^n \)
    \end{romanlist}
    \item Use o teste de Leibniz para mostrar que a sûˋrie
    \( \sum_{n=1}^\infty(-1)^{n+1}a_n \)
    converge.
    \item Mostre que o teste da raiz nûÈo ûˋ conclusivo para a sûˋrie
    \[
      \sum_{n=1}^\infty \left(\frac{a_n}{b_n}\right)^n =
      \sum_{n=1}^\infty \left[n^2\ln\left(1+\frac{1}{n^2}\right)\right]^n
    \]
    Por que o teste da raiz ûˋ inconclusivo neste caso? \\
    \hint{Dûˆ exemplos de duas sûˋries que satisfazem $\sqrt[n]{a_n}\rightarrow 1$,
      onde $a_n$ ûˋ o termo geral, mas uma converge e a outra diverge.}
  \end{alphalist}
\begin{answer}
  \begin{alphalist}
    \item
    \begin{minipage}[t]{0.9\linewidth}
      \begin{romanlist}
        \item \(\sum_{n=1}^\infty b_n=\sum_{n=1}^\infty \frac{1}{n^2}\)
        \item \(\sum_{n=1}^\infty a_n=\sum_{n=1}^\infty \ln\left(1+\frac{1}{n^2}\right)\)
        \item \(\sum_{n=1}^\infty \frac{a_n}{b_n}=\sum_{n=1}^\infty n^2\ln\left(1+\frac{1}{n^2}\right)\)
        \item \(\sum_{n=1}^\infty a_n^n=\sum_{n=1}^\infty \left[\ln\left(1+\frac{1}{n^2}\right)\right]^n\)
      \end{romanlist}
    \end{minipage}
%    \item ~
%    \item ~
  \end{alphalist}
\end{answer}
\end{exercise}
