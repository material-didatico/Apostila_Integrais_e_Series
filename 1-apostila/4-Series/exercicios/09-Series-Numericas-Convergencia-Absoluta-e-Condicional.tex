\begin{exercise}
  Determine se as séries convergem absolutamente, convergem ou divergem.
  \begin{mathlist}
    % 15 - 20
    \item \sum_{n=1}^\infty (-1)^{n+1} (0,1)^n
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{(0,1)^n}{n}
    \item \sum_{n=1}^\infty \frac{(-1)^{n}}{\sqrt{n}}
    \item \sum_{n=1}^\infty \frac{(-1)^{n}}{1+\sqrt{n}}
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{n}{n^3+1}
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{n!}{2^n}
    % 21 - 26
    \item \sum_{n=1}^\infty (-1)^{n}   \frac{1}{n+3}
    \item \sum_{n=1}^\infty (-1)^{n}   \frac{\sin(n)}{n^2}
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{3+n}{5+n}
    \item \sum_{n=1}^\infty \frac{(-2)^{n+1}}{n+5^n}
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{1+n}{n^2}
    \item \sum_{n=1}^\infty (-1)^{n+1}\, \sqrt[n]{10}
  \end{mathlist}
\end{exercise}

\begin{exercise}
  Determine se as séries convergem absolutamente, apenas convergem ou divergem.
  \begin{mathlist}[1]
    % 27 - 34
    \item \sum_{n=1}^\infty (-1)^{n} n^2 \left(\frac{2}{3}\right)^n
    \item \sum_{n=1}^\infty \frac{(-1)^{n+1}}{n\ln(n)}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{\arctan(n)}{n^2+1}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{\ln(n)}{n-\ln(n)}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{n}{n+1}
    \item \sum_{n=1}^\infty (-5)^{-n}
    \item \sum_{n=1}^\infty \frac{(-100)^{n}}{n!}
    \item \sum_{n=1}^\infty \frac{(-1)^{n+1}}{n^2+2n+1}
    % 35 - 42
    \item \sum_{n=1}^\infty \frac{\cos(n\pi)}{n\sqrt{n}}
    \item \sum_{n=1}^\infty \frac{\cos(n\pi)}{n}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{(n+1)^n}{(2n)^n}
    \item \sum_{n=1}^\infty (-1)^{n+1} \frac{(n!)^2}{(2n)!}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{(2n)!}{2^n\,n!\,n}
    \item \sum_{n=1}^\infty (-1)^{n} \frac{(n!)^2\,3^n}{(2n+1)!}
    \item \sum_{n=1}^\infty (-1)^{n} \left( \sqrt{n+1}-\sqrt{n}\right)
    \item \sum_{n=1}^\infty (-1)^{n} \left( \sqrt{n^2+n}-n\right)
  \end{mathlist}
\end{exercise}
