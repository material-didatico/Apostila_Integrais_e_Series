\begin{exercise}
  Determine se as séries satisfazem as condições para serem
  séries alternadas e se convergem ou divergem.
\begin{mathlist}[1][series=03-08-A]
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{1}{\sqrt{n}}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{1}{n^{\sfrac{3}{2}}}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{1}{3^n\,n}
  \item \sum_{n=2}^\infty (-1)^{n}   \frac{4}{(\ln n)^2}
  \item \sum_{n=1}^\infty (-1)^{n}   \frac{n}{n^2+1}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{n^2+5}{n^2+4}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{2^n}{n^2}
  \item \sum_{n=1}^\infty (-1)^{n}   \frac{10^n}{(n+1)!}
  \item \sum_{n=1}^\infty (-1)^{n+1} \left(\frac{n}{10}\right)^n
  \item \sum_{n=2}^\infty (-1)^{n+1} \frac{1}{\ln(n)}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{\ln n}{n}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{\sqrt{n}+1}{n+1}
\end{mathlist}
\begin{mathlist}[1][resume=03-08-A]
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{3\sqrt{n+1}}{\sqrt{n}+1}
  \item \sum_{n=1}^\infty (-1)^{n}   \ln\left(1+\frac{1}{n}\right)
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{3\sqrt{n+1}}{\sqrt{n}+1}
\end{mathlist}
\end{exercise}

% Ex 49-52 pg 42 Thomas
%------------------------------------------------------------------------------%
\begin{exercise}
Estime a magnitude do erro quando aproximamos o valor da soma de
cada série pela soma dos seus quatro primeiros termos.
\begin{mathlist}[2]
  \item \sum_{n=1}^\infty \frac{(-1)^{n+1}}{n}
  \item \sum_{n=1}^\infty \frac{(-1)^{n+1}}{10^n}
  \item \sum_{n=1}^\infty (-1)^{n+1} \frac{(0,01)^n}{n}
  \item \sum_{n=0}^\infty (-1)^{n}\,t^n \) \\ onde \(0<t<1\)
\end{mathlist}
\end{exercise}

% Ex 53-56 pg 43 Thomas
%------------------------------------------------------------------------------%
\begin{exercise}
  Determine quantos termos devem ser somados para aproximar a soma da série
  com erro menor do que 0,001.
\begin{mathlist}[2]
  \item \sum_{n=1}^\infty \frac{(-1)^{n}}{n^2+3}
  \item \sum_{n=1}^\infty \frac{(-1)^{n+1}\,n}{n^2+1}
  \item \sum_{n=1}^\infty \frac{(-1)^{n+1}}{\big(n+3\sqrt{n}\big)^3}
  \item \sum_{n=1}^\infty \frac{(-1)^{n}}{\ln\big(\ln(n+2)\big)}
\end{mathlist}
\end{exercise}
