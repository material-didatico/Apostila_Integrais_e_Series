\begin{exercise}
  Utilize o teste da raiz para determinar se cada série converge ou diverge.
  \begin{mathlist}
  \item \sum_{n=1}^\infty \frac{7}{(2n+5)^n}
  \item \sum_{n=1}^\infty \frac{4^n}{(3n)^n}
  \item \sum_{n=1}^\infty \left(\frac{4n+3}{3n-5}\right)^n
  \item \sum_{n=1}^\infty \left[ \ln\left(e^2+\frac{1}{n}\right) \right]^{n+1}
  \item \sum_{n=1}^\infty \frac{8}{\left(3+\sfrac{1}{n}\right)^{2n}}
  \item \sum_{n=1}^\infty \sin^n \left( \frac{1}{\sqrt{n}} \right)
  \item \sum_{n=1}^\infty \left( 1 - \frac{1}{n}\right)^{n^2} \) \\
        \hspace{10mm}\hint{Use \(\lim_{n\to\infty} \left(1+\frac{x}{n}\right)^n = e^x \)}
  \item \sum_{n=2}^\infty \frac{1}{n^{1+n}}
  \end{mathlist}
\end{exercise}

\begin{exercise}
Determine se a série converge ou diverge.
\begin{mathlist}
  % 27-31
  \item \sum_{n=1}^\infty \frac{\ln(n)}{n^3}
  \item \sum_{n=1}^\infty \frac{\big(\ln(n)\big)^n}{n^n}
  \item \sum_{n=1}^\infty \left(\frac{1}{n}-\frac{1}{n^2}\right)
  \item \sum_{n=1}^\infty \left(\frac{1}{n}-\frac{1}{n^2}\right)^n
  \item \sum_{n=1}^\infty \frac{\ln(n)}{n}
  % 39-42
  \item \sum_{n=2}^\infty \frac{n}{\big(\ln(n)\big)^n}
  \item \sum_{n=2}^\infty \frac{n}{\big(\ln(n)\big)^{\sfrac{n}{2}}}
  \item \sum_{n=1}^\infty \frac{n!\ln(n)}{n(n+2)!}
\end{mathlist}
\end{exercise}
