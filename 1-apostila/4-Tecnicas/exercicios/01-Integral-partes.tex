\begin{exercise}
Calcule as integrais, usando integração por partes
\begin{mathlist}
  \item \int x \cos(5x)\,dx
  \item \int xe^{-x}\,dx
  \item \int t^2 \sin(\beta t) \,dt\) onde $\beta$ é uma constante.
  \item \int (x^2+2x) \cos(x) \,dx
  \item \int p^5 \ln(p) \,dp
  \item \int \big(\ln(x)\big)^2 \,dx
  \item \int t \sec^2(2t) \,dt
  \item \int e^{2x} \sin(3x) \,dx
  \item \int_4^9 \frac{\ln(y)}{\sqrt{y}}\,dy
  \item \int_0^{\frac{1}{2}} x \cos(\pi x)\,dx
  \item \int_0^1 \left( x^2+1 \right) e^{-x} \,dx
  \item \int_1^2 x^4 \big(\ln(x)\big)^2 \,dx
  \item \int_1^{\sqrt{3}} \arctan\left(\frac{1}{x}\right)\,dx
  \item \int x \tan^2(x) \,dx\). Dica: $\tan^2(x)=1-\sec^2(x)
\end{mathlist}
\begin{answer}
  \begin{mathlist}
    \item \frac{1}{5}x \sin(5x)+\frac{1}{25} \cos(5x)+c
    \item -xe^{-x}-e^{-x}+C
    \item - \frac{t^2}{\beta} \cos(\beta t)
          + \frac{2t}{\beta^2} \sin(\beta t)
          + \frac{2}{\beta^3}  \cos(\beta t)
          + c
    \item \left(x^2+2x\right) \sin(x)+(2x+2)\sin(x)-2 \sin(x)+C
    \item \frac{1}{6}p^6 \ln(p) + \frac{1}{36}p^6 + C
    \item x\big(\ln(x)\big)^2 - 2x \ln(x) + 2x + c
    \item \frac{t}{2} \tan(2t) - \frac{1}{4} \ln\abs{\sec(x)} + C
    \item \frac{1}{13}e^{2x} \left( 2 \sin(3x)-3 \cos(3x) \right) + C
    \item  6\ln(9) - 4\ln(4) - 4
    \item \frac{\pi-2}{\pi^2}
    \item -\frac{6}{e} + 3
    \item \frac{32}{5}\big(\ln(2)\big)^2 - \frac{64}{25}\ln(2) + \frac{62}{126}
    \item \frac{\pi \sqrt{3}}{6}- \frac{\pi}{4} + \frac{1}{2}\ln(2)
    \item x \tan(x)- \ln\abs{\sec(x)} - \frac{1}{2}x^2 + C
  \end{mathlist}
\end{answer}
\end{exercise}

\begin{exercise}
  Encontre a área sob a curva $y=x \sin(x)$ de 0 a $\pi$.
  \begin{answer}
    $A=\pi$
  \end{answer}
\end{exercise}

\begin{exercise}
  Uma partícula se move ao longo de uma reta com velocidade igual a
  $v(t)=t^2e^{-t}$ metros por segundo após $t$ segundos.
  Qual a distância que essa partícula percorrerá nos primeiros $t$ segundos?
  \begin{answer}
    $s=2-e^{-t}\left(t^2+2t+2\right)$ \si{meter}
  \end{answer}
\end{exercise}

\begin{exercise}
  Um arquiteto propôs colocar no piso de uma praça pastilhas coloridas
  em uma região plana cujo o formato se assemelha (respeitando a escala)
  a região delimitada simultaneamente pelas quarto curvas
  $y= \ln(x)$, $y=e^x$, $y=1-x$ e $x=e$.
  Esboce a região e calcule a área.
\end{exercise}

\begin{exercise}
Calcule a área entre as curvas $y=x^2e^{-x}$ e $y=xe^{-x}$.
\begin{answer}
  $A=3e^{-1}-1$
\end{answer}
\end{exercise}

\begin{exercise}
Calcule a integral imprópria \(\int_0^{\infty} x^2e^{-x}dx\).
\begin{answer}
A integral é convergente e vale 2.
\end{answer}
\end{exercise}

\begin{exercise}
Calcule o volume do sólido gerado pela rotação da região do primeiro quadrante
delimitada pelos eixos coordenados, pela curva $y=e^x$ e pela reta $x=\ln(2)$
em torno da reta $x=\ln(2)$.
\begin{answer}
\(2\pi\big(1-\ln(2)\big)\).
\end{answer}
\end{exercise}

\begin{exercise}
  Considere a região delimitada pelos gráficos de $y=\ln(x)$
  e pela reta $x=\ln(2)$, $y=0$ e $x=e$.
  \begin{alphalist}
      \item Determine a área dessa região.
      \item Determine o volume do sólido gerado pela rotação dessa
            região em torno do eixo das abscissas.
      \item Determine o volume do sólido gerado pela rotação dessa
            região em torno da reta $x=-2$.
  \end{alphalist}
  \begin{answer}
    \begin{mathlist}[2]
      \item 1
      \item (e-2)\pi
      \item \frac{\pi}{2}\left(e^2+9\right)
    \end{mathlist}
  \end{answer}
\end{exercise}

\begin{exercise}
  Encontre o valor médio de $f(x)= x^2\ln(x)$ no intervalo $[1,3]$.
  \begin{answer}
  $\frac{9}{2}\ln(3)- \frac{13}{9}$
  \end{answer}
\end{exercise}
