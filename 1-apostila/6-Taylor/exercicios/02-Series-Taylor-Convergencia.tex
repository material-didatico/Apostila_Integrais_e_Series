\begin{exercise}
  Encontre a série de Taylor de $\cos(x)$ centrada em $x=0$,
  mostre que o $\cos(x)$ coincide com a série obtida para todo $x\in\R$.
\end{exercise}

\begin{exercise}
	Calcule a série de Maclaurin ou Taylor da função, determine seu raio de convergência
	e prove que a série converge para a função no intervalo de convergência.
	\begin{mathlist}
		\item f(x) = \sin(\pi x)
		\item f(x) = \sin(x) \qquad a = \sfrac{\pi}{2}
		\item f(x) = \sinh(x)
		\item f(x) = \cosh(x)
	\end{mathlist}
\end{exercise}
