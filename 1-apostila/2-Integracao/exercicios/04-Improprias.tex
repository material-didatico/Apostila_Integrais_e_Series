\begin{exercise}
Determine se cada integral é convergente ou divergente.
Calcule aquelas que são convergentes.
\begin{mathlist}
  \item \int_{1}^{\infty} \frac{1}{(x-2)^{\frac{3}{2}}} dx
  \item \int_{2}^{\infty} e^{-5p}dp
  \item \int_{0}^{\infty} \frac{x^2}{\sqrt{1+x^3}} dx
  \item \int_{-\infty}^{\infty} xe^{-x^2} dx
  \item \int_{-\infty}^{0} ze^{2z}dz
  \item \int_{1}^{\infty} \frac{ln(x)}{x} dx
  \item \int_{0}^{\infty} \sin^2(x) dx
  \item \int_{0}^{\infty} \frac{1}{v^2+2v-3} dv
  \item \int_{0}^{1} \frac{3}{x^5} dx
  \item \int_{2}^{3} \frac{1}{\sqrt{3-x}} dx
  \item \int_{-2}^{3} \frac{1}{x^4} dx
  \item \int_{0}^{3} \frac{dx}{x^2-6x+5} dx
  \item \int_{-1}^{0} \frac{e^{\frac{1}{x}}}{x^2} dx
\end{mathlist}
\begin{answer}
\begin{alphalist}
  \item converge para 2
  \item converge para \(\frac{1}{5} e^{-10}\)
  \item divergente
  \item converge para 0
  \item converge para \(-\frac{1}{4}\)
  \item divergente
  \item divergente
  \item converge para \(\frac{1}{4} \ln 5\)
  \item divergente
  \item Converge para 2
  \item Divergente
  \item Divergente
  \item Converge para \(\frac{-2}{e}\)
\end{alphalist}
\end{answer}
\end{exercise}
