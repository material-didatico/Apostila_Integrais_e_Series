\begin{exercise}
Determine se cada integral é convergente ou divergente.
Calcule aquelas que são convergentes.
\begin{mathlist}
  \item \int_{1}^{\infty} \frac{1}{(x-2)^{\frac{3}{2}}} dx
  \item \int_{2}^{\infty} e^{-5p}dp
  \item \int_{0}^{1} \frac{3}{x^5} dx
  \item \int_{2}^{3} \frac{1}{\sqrt{3-x}} dx
  \item \int_{-2}^{3} \frac{1}{x^4} dx
  \item \int_{-\infty}^{-1} \frac{1}{\sqrt{2-x}}dx
  \item \int_{-\infty}^{\infty} (2-v^4)dv
\end{mathlist}
\begin{answer}
\begin{alphalist}
  \item converge para 2
  \item converge para \(\frac{1}{5} e^{-10}\)
  \item divergente
  \item converge para 2
  \item divergente
  \item divergente
  \item divergente
\end{alphalist}
\end{answer}
\end{exercise}

\begin{exercise}
  Esse exercício tem como objetivo mostrar que não podemos definir
  \[
    \int_{-\infty}^{\infty}f(x)dx
    = \lim_{t \to \infty} \int_{-t}^t f(x)dx \,.
  \]
  Para isso:
  \begin{alphalist}
   \item Mostre que \(\int_{-\infty}^{\infty} x dx\) é divergente.
   \item Mostre que \(\lim_{t \to \infty} x=0\)
  \end{alphalist}
\end{exercise}
