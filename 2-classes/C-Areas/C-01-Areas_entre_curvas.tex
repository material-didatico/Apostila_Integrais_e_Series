\section{Áreas entre Curvas}

%------------------------------------------------------------------------------%
\framedgraphic{Área entre Curvas}{area_entre_curvas_01}{}

%------------------------------------------------------------------------------%
\begin{frame}{Definição: Área entre Curvas}
  A área da região limitada pelas curvas e retas
  \[
    \pause
    y = f(x)  \qquad
    \pause
    y = g(x)  \qquad
    \pause
    x = a     \qquad
    \pause
    x = b
  \]
  \pause
  onde $f$ e $g$ são contínuas e
  \pause
  $f(x) \geq g(x)$ para todo $x \in [a, b]$,
  \pause
    é
  \[
    A = \int_a^b f(x) - g(x) \, dx
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Passo a Passo}
  \begin{enumerate}[<+->]
    \item Identificar a região
    \item Dividir a região se necessário
    \item Identificar qual é a função maior em cada parte da região
    \item Montar a integral definida que representa a área de cada parte
    \item Calcular as integrais definidas
    \item Somar os resultados
  \end{enumerate}
\end{frame}

\subsection{Exemplo -- 1}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 1}
  Encontre a área da região limitada abaixo por $f(x)= x$, \\
  acima por $g(x)=e^x+1$ e nos lados pelas retas $x=0$ e $x=1$
\end{frame}

%\framedgraphic{Exemplo 1 -- Interpretação}{areae1}{}
\begin{frame}{Exemplo 1 -- Interpretação}
  \centering
  \includegraphics[trim=40mm 28mm 40mm 0mm, clip,
  width=\textwidth,
  height=0.8\textheight,
  keepaspectratio]{areae1}
\end{frame}

\begin{frame}{Exemplo 1 -- Resolução}
  \begin{align*}
    \onslide<+->{ A & = \int_0^1 g(x) - f(x) dx \\[2mm] }
    \onslide<+->{   & = \int_0^1 e^x + 1 - x \, dx \\[2mm]}
    \onslide<+->{   & = \left( e^x+x-\frac{x^2}{2} \right) \evalat{0}{1} & & \text{TFC 2}\\[2mm]}
    \onslide<+->{   & = \left(e+1-\frac{1}{2}\right)-\left(e^0+0-\frac{0}{2}\right) \\[2mm]}
    \onslide<+->{   & = e-\frac{1}{2}}
  \end{align*}
\end{frame}

\subsection{Exemplo -- 2}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 2}
  Calcule a área da região delimitada pelas curvas $y = x^2$ e $y = 2x$
\end{frame}

\begin{frame}{Exemplo 2 -- Interpretação}
  Observe que:
  \begin{itemize}[<+->]
    \item a curva \( y = x^2 \) é uma parábola
    \item a curva \( y = 2x  \) é uma reta que passa pela origem
    \item \( y = 2x  \) está ``por cima''
    \item \( y = x^2 \) está ``por baixo''
    \item os extremos do intervalo precisam ser determinados
  \end{itemize}
  \addgraphic{5}{10.5,2.5}{areae2}
\end{frame}

\begin{frame}{Exemplo 2 -- Resolução}
  \onslide<+->{Precisamos encontrar os pontos de intersecção das curvas}
  \begin{align*}
    \onslide<+->{x^2        & = 2x \\}
    \onslide<+->{x^2 - 2x   & = 0  \\}
    \onslide<+->{x(x-2)     & = 0}
  \end{align*}
  \onslide<+->{Soluções: $x=0$ e $x=2$}
\end{frame}

\begin{frame}{Exemplo 2 -- Resolução}
  \begin{align*}
    \onslide<+->{ A & = \int_0^2 2x - x^2 \, dx                                    \\[2mm]}
    \onslide<+->{   & = \left( x^2-\frac{x^3}{3} \right) \evalat{0}{2}             \\[2mm]}
    \onslide<+->{   & = \left( 4- \frac{8}{3}\right) - \left( 0-\frac{0}{3}\right) \\[2mm]}
    \onslide<+->{   & = \frac{12-8}{3}                                             \\[2mm]}
    \onslide<+->{   & = \frac{4}{3}                                                       }
  \end{align*}
\end{frame}

\begin{frame}{Exemplo 3}

  \[ y = -x^2 + 3x + 2\]
  e
  \[ y = x^2 - 3 \]

\begin{frame}{Exemplo 3 -- Interpretação}
  \(y = -x^2 +3x + 2\) é uma parábola com concavidade voltada para baixo

  \pause
  Vértice
  \[
    \pause
    \Delta
    = b^2 - 4ac
    \pause
    = 3^2 - 4(-1)2
    \pause
    = 9 + 8
    \pause
    = 17
  \]
  \[
    \pause
    x_v
    = \frac{-b}{2a}
    \pause
    = \frac{-3}{2(-1)}
    \pause
    = \frac{3}{2}
  \qquad\qquad
    \pause
    y_v
    = \frac{-\Delta}{4a}
    \pause
    = \frac{17}{4}
  \]
  \pause
  Raízes
  \[
    \pause
    x_1
    = \frac{-b-\sqrt{\Delta}}{2a}
    \pause
    = \frac{3+\sqrt{17}}{2}
    \qquad\qquad
    \pause
    x_2
    = \frac{-b+\sqrt{\Delta}}{2a}
    \pause
    = \frac{3-\sqrt{17}}{2}
  \]
  \pause
  Intercepta o eixo $y$ no ponto $(0, 2)$
\end{frame}

\begin{frame}{Exemplo 3 -- Interpretação}

  \pause
  Vértice

    \Delta
    = b^2 - 4ac
    \pause
    = 0^2 - 4\times 1\times (-3)
    \pause
    = 12
  \]
  \[
    \pause
    x_v
    = \frac{-b}{2a}
    \pause
    = \frac{0}{2}
    \pause
    = 0
  \qquad\qquad
    \pause
    y_v
    = \frac{-\Delta}{4a}
    \pause
    = \frac{-12}{4}
    \pause
    = -3
  \]
  \pause
  Raízes
  \[
    \pause
    x_1
%    = \frac{-b-\sqrt{\Delta}}{2a}
%    \pause
    = \frac{-\sqrt{12}}{2}
    \pause
    = \frac{-\sqrt{4}\sqrt{3}}{2}
    \pause
    = -\sqrt{3}
    \qquad\qquad
    \pause
    x_2
%    = \frac{-b+\sqrt{\Delta}}{2a}
%    \pause
    = \frac{\sqrt{12}}{2}
    \pause
    = \sqrt{3}
  \]
  \pause
  Intercepta o eixo $y$ no ponto $(0, -3)$
\end{frame}

\begin{frame}{Exemplo 3 -- Interpretação}
  \centering

\begin{frame}{Exemplo 3 -- Resolução}
  Pontos de intersecção
  \pause
  \begin{align*}
    -x^2 +3x + 2 & = x^2 - 3 \\
    2x^2 -3x - 5 & = 0
  \end{align*}
  \[
    \pause
    \Delta
    \pause
    = b^2 - 4ac
    \pause
    = (-3)^2 - 4\times 2 \times (-5)
    \pause
    = 9 + 40
    \pause
    = 49
  \]
  \pause
  Soluções
  \[
    \pause
    x_1
    = \frac{3-\sqrt{49}}{2\times 2}
    \pause
    = \frac{3-7}{4}
    \pause
    = -1
    \qquad\qquad
    \pause
    x_2
    = \frac{3+\sqrt{49}}{2\times 2}
    \pause
    = \frac{3+7}{4}
    \pause
    = \frac{5}{2}
  \]
\end{frame}

\begin{frame}{Exemplo 3 -- Resolução}
  \begin{align*}
    \onslide<+->{
      A
      & = \int_{-1}^{\sfrac{5}{2}} \left(-x^2 + 3x +  2\right) - \left( x^2 -3 \right) \, dx \\[2mm]}
    \onslide<+->{
      & = \int_{-1}^{\sfrac{5}{2}} -2x^2 +3x + 5  \, dx  \\[2mm]}
    \onslide<+->{
      & = \left( -2\frac{x^3}{3} -3\frac{x^2}{2} +5x \right) \evalat{-1}{\sfrac{5}{2}} \\[2mm]}
    \onslide<+->{
      & = \left(
        - \frac{2}{3} \frac{125}{8}
        + \frac{3}{2} \frac{25}{4}
        + 5 \frac{5}{2}
      \right)
      - \left(
        - \frac{2}{3} (-1)^3
        + \frac{3}{2} (-1)^2
        + 5 (-1)
      \right) \\[2mm]}
      \onslide<+->{
      & = \frac{343}{24}}
  \end{align*}
\end{frame}

\subsection{Exemplo -- 4}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 4}
  Calcule a área da região delimitada pelas curvas
  \[
    f(x) = \sin(x)
    \quad\text{e}\quad
    g(x) = \cos(x)
  \]
  no intervalo
  \([0, \sfrac{\pi}{2}]\)
\end{frame}

\framedgraphic{Exemplo 4 -- Interpretação}{areae4}{}

\begin{frame}{Exemplo 4 -- Interpretação}
  Pontos de interseção
  \pause
  \[
    \sin(x) = \cos(x) \qquad 0 \leq x \leq \sfrac{\pi}{2}
  \]
  \[
    \pause
    x = \sfrac{\pi}{4}
  \]

  \pause
  Pelo gráfico podemos notar que
  \[
    \pause
    \cos(x) \geq \sin(x) \qquad 0 \leq x \leq \sfrac{\pi}{4}
  \]
  \[
    \pause
    \sin(x) \geq \cos(x) \qquad \sfrac{\pi}{4} \leq x \leq \sfrac{\pi}{2}
  \]

  \pause
  Precisamos dividir a região em duas partes
\end{frame}

\begin{frame}{Exemplo 4 -- Resolução}
  \begin{align*}
    \onslide<+->{A & = A_1 + A_2                                                                 \\[2mm]}
    \onslide<+->{  & = \int_0^{\sfrac{\pi}{4}}                \cos(x) - \sin(x) \,dx
                     + \int_{\sfrac{\pi}{4}}^{\sfrac{\pi}{2}} \sin(x) - \cos(x) \,dx             \\[2mm]}
    \onslide<+->{  & = \big(  \sin(x) + \cos(x) \big) \evalat{0}{\sfrac{\pi}{4}}
                     + \big( -\cos(x) - \sin(x) \big) \evalat{\sfrac{\pi}{4}}{\sfrac{\pi}{2}}    \\[2mm]}
    \onslide<+->{  & = \left( \frac{\sqrt{2}}{2} + \frac{\sqrt{2}}{2} -0 -1 \right)
                     + \left( -0 -1 + \frac{\sqrt{2}}{2} + \frac{\sqrt{2}}{2} \right)            \\[2mm]}
    \onslide<+->{  & = 2\sqrt{2}-2}
  \end{align*}
\end{frame}

\begin{frame}{Simetrias}
  Poderíamos ter aproveitado a
  simetria da região em relação à reta $x = \sfrac{\pi}{4}$,
  o que nos permitiria economizar esforço ao calcular a área, uma vez que
  \[
    A
    = 2 A_1
    = 2 \int_0^{\sfrac{\pi}{4}} \cos(x) - \sin(x) \,dx
  \]
\end{frame}

\subsection{Exemplo -- 5}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 5}
  Calcule a área da região expressa na figura, que é uma das regiões
  delimitada simultaneamente pelas três curvas
  \[
    f(x) = 2x-1, \qquad
    g(x) =-\frac{x}{2}+4\quad\text{e}\quad
    h(x)=(x-2)^2
  \]
  \centerline{\includegraphics[height=50mm]{areae51}}
\end{frame}

\begin{frame}{Exemplo 5 -- Interpretação}
  \(f(x)\) e \(g(x)\) são retas

  \pause
  \vspace{0.5\baselineskip}
  \(h(x)=(x-2)^2=x^2-4x+4\) \; parábola com concavidade para cima

  \pause
  \vspace{0.5\baselineskip}
  Vértice $(2, 0)$

  \pause
  \vspace{0.5\baselineskip}
  Raiz $x = 2$
\end{frame}

\begin{frame}{Exemplo 5 -- Interpretação}
  Precisamos dividir a região em duas

  \pause
  \vspace{0.5\baselineskip}
  De 0 até 1                                   \\[2mm]
  \(g(x) = -\sfrac{x}{2}+4\) \tabto{37mm} está ``por cima'' \\[2mm]
  \pause
  \(h(x) = (x-2)^2\)         \tabto{37mm} está ``por baixo''

  \pause
  \vspace{0.5\baselineskip}
  De 1 até 2                                    \\[2mm]
  \(g(x) = -\sfrac{x}{2}+4\) \tabto{37mm} está ``por cima'' \\[2mm]
  \pause
  \(f(x) = 2x-1\)            \tabto{37mm} está ``por baixo''
\end{frame}

\begin{frame}{Exemplo 5 -- Resolução}
  \begin{align*}
    \onslide<+->{A_1 & = \int_0^1 g(x) - h(x) dx \\[5mm]}
    \onslide<+->{A_2 & = \int_1^2 g(x) - f(x) dx}
  \end{align*}
\end{frame}

\begin{frame}{Exemplo 5 -- Resolução}
  \begin{align*}
    \onslide<+->{A_1 & = \int_0^1 \left(-\frac{x}{2}+4\right) - \left(x^2-4x+4\right) \, dx \\[2mm]}
    \onslide<+->{    & = \int_0^1 -x^2 + \frac{7}{2}x \,  dx                        \\[2mm]}
    \onslide<+->{    & = \left( -\frac{x^3}{3}+\frac{7}{2}\frac{x^2}{2} \right) \evalat{0}{1} \\[2mm]}
    \onslide<+->{    & = \left(-\frac{1}{3}+\frac{7}{4} \right)-(0-0)               \\[2mm]}
    \onslide<+->{    & = \frac{17}{12}}
  \end{align*}
\end{frame}

\begin{frame}{Exemplo 5 -- Resolução}
  \begin{align*}
    \onslide<+->{A_2 & = \int_1^2 \left(-\frac{x}{2}+4\right)-\left(2x-1\right) \, dx                      \\[2mm]}
    \onslide<+->{    & = \int_1^2 -\frac{5}{2}x+5 \, dx                              \\[2mm]}
    \onslide<+->{    & = \left( -\frac{5}{2}\frac{x^2}{2}+5x \right) \evalat{1}{2}             \\[2mm]}
    \onslide<+->{    & = \left(-\frac{20}{4}+10 \right)-\left(-\frac{5}{4}+5 \right) \\[2mm]}
    \onslide<+->{    & = \frac{5}{4}}
  \end{align*}
\end{frame}

\begin{frame}{Exemplo 5 -- Resolução}
  \[
    A
    = A_1 + A_2
    \pause
    = \frac{17}{12} + \frac{5}{4}
    \pause
    = \frac{17+15}{12}
    \pause
    = \frac{32}{12}
    \pause
    = \frac{8}{3}
  \]
\end{frame}
