\section{Teorema do Confronto}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema do Confronto}
  Sejam $(a_k)$, $(b_k)$ e $(c_k)$ tais que
  \[
   	\memph{a_k \;\leq\; b_k \;\leq\; c_k} \qquad\qquad \forall\; k > N
 	\]

	\pause
  \vspace{\baselineskip}
  Se $\quad a_k\to L\quad$ e $\quad c_k\to L\quad$ então $\quad b_k\to L$
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema da Função Contínua}
É verdade que \quad
\emph{\(
  \lim_{k\to \infty} f(a_k) = f\!\left(\lim_{k\to \infty} a_k\right)
\)} \quad?

\pause
\vspace{2\baselineskip}
Se $a_k$ é uma sequência numérica real convergente, $a_k\to L$, e\\[1mm]
$f\colon D\subset\R\to\R$ é uma função real contínua em $D$,
com $L\in D$
então
\[
  \lim f(a_k) = f(L)
\]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  Calcular \(\lim\left[\cos\left(\alpha^k\right)\right]\) com $0<\alpha<1$

  \pause
  \vspace{\baselineskip}
  \qquad Considerando a sequência \( a_k = \alpha^k\)

  \pause
  \qquad Sabemos que \(\lim a_k=0\)

  \pause
  \qquad A função cosseno é contínua em zero

  \pause
  \vspace{0.5\baselineskip}
  Portanto
  \[
    \lim\cos(\alpha^k)
    = \lim\cos(a_k)
    \pause
    = \cos\big(\lim a_k\big)
    \pause
    = \cos(0)
    \pause
    = 1
  \]
\end{frame}

\section{Sequência Definida por Função}

%------------------------------------------------------------------------------%
\begin{frame}{Limite de sequência definida por função}
    Se $f$ é uma função real e \(a_k=f(k)\) então

    \[
    	\lim_{x\to\infty} f(x) = L \quad \Rightarrow \quad
      \lim_{k\to\infty} a_k = L
    \]

    \pause
    \vspace{\baselineskip}
    Esse resultado nos permite usar a regra de l'Hôpital (em $x$)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  Mostre que \( \lim\frac{5k^4}{k^4+8k^3} = 5 \)

  \pause
  A função
  \[
    f(x) = \frac{5x^4}{x^4+8x^3}
    \hspace{50mm}
    \emph{x\in\R}
  \]
  está definida para todo $x>0$
  \pause
  e a sequência
  \[
    a_k = f(k)=\frac{5k^4}{k^4+8k^3}
    \hspace{40.5mm}
    \emph{k\in\N}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  \onslide<+->{Podemos aplicar a regra de L'Hôpital \emph{em $x$},}
  \onslide<+->{pois temos uma indeterminação $\sfrac{\infty}{\infty}$}
  \begin{align*}
    \onslide<+->{    \lim_{k\to\infty} \frac{5k^4}{k^4+8k^3}}
    \onslide<+->{  = \lim_{x\to\infty} \frac{5x^4}{x^4+8x^3}}
    \onslide<+->{& = \lim_{x\to\infty} \frac{20x^3}{4x^3+24x^2} \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{60x^2}{12x^2+48x}  \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{120x}{24x+48}      \\[2mm]}
    \onslide<+->{& = \lim_{x\to\infty} \frac{120}{24}           \\[2mm]}
    \onslide<+->{& = 5}
  \end{align*}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo}
  \onslide<+->{Método mais apropriado}
  \begin{align*}
    \onslide<+->{    \lim_{k\to\infty} \frac{5k^4}{k^4+8k^3}}
    \onslide<+->{& = \lim_{k\to\infty} \frac{\dfrac{\,5k^4\,}{k^4}}{\;\dfrac{\;k^4+8k^3\;}{k^4}\;} \\[3mm]}
    \onslide<+->{& = \lim_{k\to\infty} \frac{5}{1+\dfrac{8}{k}} \\[2mm]}
    \onslide<+->{& = \frac{5}{\displaystyle 1 + 8\lim_{k\to\infty} \frac{1}{k}} \\[2mm]}
    \onslide<+->{& = 5}
  \end{align*}
\end{frame}

\section{Sequência Limitadas}

%------------------------------------------------------------------------------%
\begin{frame}{Sequências Limitadas}
	Uma sequência $(a_k)$ é

	\pause
  \vspace{\baselineskip}
	\emph{Limitada inferiormente} \tabto{50mm} se existe $m\in\R$ tal que \emph{$m\leq a_k$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada superiormente} \tabto{50mm} se existe $M\in\R$ tal que \emph{$M\geq a_k$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Limitada}               \tabto{50mm} se é limitada superiormente e inferiormente

\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Sequências Crescentes e Decrescentes}
	Uma sequência $(a_k)$ é

	\vspace{\baselineskip}
	\emph{Crescente}   \tabto{30mm} se \emph{$\;a_k \leq a_{k+1}\;$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Decrescente} \tabto{30mm} se \emph{$\;a_k \geq a_{k+1}\;$} para todo $k\in\N$

	\pause
	\vspace{\baselineskip}
	\emph{Monótona}    \tabto{30mm} se é crescente ou decrescente
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Convergência de sequência monótona}
	Se uma sequência $(a_k)$ é \emph{limitada e monotônica},
  então ela é \emph{convergente}

	\pause
	\vspace{2\baselineskip}
	Importante porque permite decidir a convergência sem calcular o limite
\end{frame}
