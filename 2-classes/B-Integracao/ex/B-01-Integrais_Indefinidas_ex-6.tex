\begin{question}
  Calcule a integral
  \(
    \int x^2\sin(\pi t) \, dt
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{F(t) & = \int x^2\sin(\pi t) \, dt}  \\[2mm]
    \onslide<+->{     & = x^2 \int \sin(\pi t) \, dt}  \\[2mm]
    \onslide<+->{     & = x^2 \big(-\cos(\pi t) \big) \frac{1}{\pi} + c} \\[2mm]
    \onslide<+->{     & = -\frac{x^2}{\pi} \cos(\pi t) + c}
  \end{align*}
\end{resolution}
