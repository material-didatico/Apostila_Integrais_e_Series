\begin{question}
  Calcule \(\int_{-2}^4 \abs{x} \,dx\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{A
    & = \int_{-2}^4 \abs{x} \,dx                                       \\[1mm]}
    \onslide<+->{& = \int_{-2}^0 \abs{x} \,dx + \int_0^4 \abs{x} \,dx  \\[1mm]}
    \onslide<+->{& = \int_{-2}^0 -x \,dx + \int_0^4 x \,dx             \\[1mm]}
    \onslide<+->{& = \frac{2^2}{2} + \frac{4^2}{2}                     \\[1mm]}
    \onslide<+->{& = 2 + 8                                             \\[1mm]}
    \onslide<+->{& = 10}
  \end{align*}
\end{resolution}
