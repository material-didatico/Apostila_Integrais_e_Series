%------------------------------------------------------------------------------%
\begin{question}
  Determine se a integral
  \[
    \int_{-1}^{0} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx
  \]
  é convergente ou divergente
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Queremos integrar
  \(f(x) = \frac{1}{\,x^{\sfrac{2}{3}}}\)
  no intervalo \([-1, 0]\)

  \pause
  \smallskip
  Mas $f$ é descontínua em $x=0$

  \pause
  Não podemos usar o Teorema Fundamental do Cálculo

  \pause
  Essa é uma integral imprópria,
  \pause
  portanto
  \[
    \int_{-1}^{0} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx
    = \lim_{t \to 0^{-}} \int_{-1}^{t} x^{\sfrac{-2}{3}}\;dx
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{\int_{-1}^{0} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx}
    \onslide<+->{& = \lim_{t \to 0^{-}} \int_{-1}^{t} x^{\sfrac{-2}{3}}\;dx  \\}
    \onslide<+->{& = \lim_{t \to 0^{-}} \left(\frac{x^{\sfrac{-2}{3}+1}}{\sfrac{-2}{3}+1} \evalat{-1}{t}\right) \\}
    \onslide<+->{& = \lim_{t \to 0^{-}} \left(3x^{\sfrac{1}{3}} \evalat{-1}{t}\right) \\}
    \onslide<+->{& = \lim_{t \to 0^{-}} 3 \left[ t^{\sfrac{1}{3}} - (-1)^{\sfrac{1}{3}} \right] \\}
    \onslide<+->{& = 3}
    \onslide<+->{& \text{Portanto, a integral imprópria converge}}
  \end{align*}
\end{resolution}
