%------------------------------------------------------------------------------%
\begin{question}
  Determine se a integral
  \[
    \int_{-1}^{0} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx
  \]
  é convergente ou divergente
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Queremos integrar
  \(f(x) = \frac{1}{\,x^{\sfrac{2}{3}}}\)
  no intervalo \([-1, 0]\)

  \pause
  \smallskip
  Mas $f$ é descontínua em $x=0$

  \pause
  Não podemos usar o Teorema Fundamental do Cálculo

  \pause
  Essa é uma integral imprópria,
  \pause
  portanto
  \[
    A
    = \int_{-1}^{0} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx
    = \lim_{t \to 0^{-}} \int_{-1}^{t} x^{\sfrac{-2}{3}}\;dx
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Primitiva}
  \begin{align*}
    \onslide<+->{F(x)}
    \onslide<+->{& = \int \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx           \\}
    \onslide<+->{& = \int x^{\sfrac{-2}{3}}\;dx                        \\}
    \onslide<+->{& = \frac{x^{\sfrac{-2}{3}+1}}{\sfrac{-2}{3}+1} + c   \\}
    \onslide<+->{& = \frac{x^{\sfrac{1}{3}}}{\sfrac{1}{3}} + c         \\}
    \onslide<+->{& = 3\,\sqrt[3]{x} + c }
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Integral definida}
  \begin{align*}
    \onslide<+->{I(t) & = \int_{-1}^{t} \frac{1}{\,x^{\sfrac{2}{3}}\,}\;dx   \\}
    \onslide<+->{& = F(x) \evalat{-1}{t}                                     \\}
    \onslide<+->{& = 3\,\sqrt[3]{x} \;\evalat{-1}{t}                         \\}
    \onslide<+->{& = 3\,\sqrt[3]{t} -3\,\sqrt[3]{-1}                         \\}
    \onslide<+->{& = 3\,\sqrt[3]{t} +3}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando o limite}
  \begin{align*}
    \onslide<+->{A & = \lim_{t \to 0^{-}} I(t)  \\}
    \onslide<+->{  & = \lim_{t \to 0^{-}} \left(3\,\sqrt[3]{t} +3\right) \\}
    \onslide<+->{  & = 3\lim_{t \to 0^{-}} \sqrt[3]{t} + \lim_{t \to 0^{-}} 3 \\}
    \onslide<+->{  & = 0 + 3 \\}
    \onslide<+->{  & = 3}
  \end{align*}
  \onslide<+->{Portanto, a integral imprópria converge}
\end{resolution}
