%------------------------------------------------------------------------------%
\begin{question}
  Calcule a área abaixo do gráfico da função \(f(x) = x + 2\) no intervalo
  \([0, b]\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{A & = \int_0^b f(x)\,dx    \\[2mm]}
    \onslide<+->{& = \int_0^b x + 2 \,dx    \\[2mm]}
    \onslide<+->{& = \int_0^b x\, dx + 2 \int_0^b dx
                 & & \text{(Linearidade)}   \\[2mm]}
    \onslide<+->{& = \frac{b^2}{2} + 2b
                 & & \text{(Área do triângulo e do retângulo)}}
  \end{align*}
\end{resolution}
