\begin{question}
  Calcule \(\int_{-7}^7 x^3 dx\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    A
    = \int_{-7}^7 x^3 dx
    \pause
    = \int_{-7}^0 x^3 dx + \int_0^7 x^3 dx
  \]
  \pause
  por simetria
  \[
    \int_{-7}^0 x^3 dx = - \int_0^7 x^3 dx
  \]
  \pause
  portanto
  \[
    A = 0
  \]
\end{resolution}
