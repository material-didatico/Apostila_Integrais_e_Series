%------------------------------------------------------------------------------%
\begin{question}
  Calcule \(\int_3^6 \frac{dx}{x}\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \(f(x)=\frac{1}{x}\) é contínua no intervalo $[3, 6]$

  \pause
  \bigskip
  podemos usar o Teorema Fundamental do Cálculo
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Primitiva}
  \onslide<+->{Calculando uma primitiva de $f$ }
  \begin{align*}
    \onslide<+->{F(x)}
    \onslide<+->{& = \int f(x) dx        \\[2mm]}
    \onslide<+->{& = \int \frac{1}{x} dx \\[2mm]}
    \onslide<+->{& = \ln\abs{x} + c}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Pelo Teorema Fundamental do Cálculo
  \[
    \pause
    \int_3^6 \frac{dx}{x}
    \pause
    \;=\; \ln\abs{x} \;\evalat{1}{3}
    \pause
    \;=\; \ln(6) - \ln(3)
    \pause
    \;=\; \ln\left(\frac{6}{3}\right)
    \pause
    \;=\; \ln(2)
    \pause
    {\color{gray}\;\approx\; 0.693}
  \]
\end{resolution}
