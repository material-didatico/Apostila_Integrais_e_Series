\begin{question}
  Calcule a derivada da função \(g(x)=\int_0^x \sqrt{1+t^2} \,dt\)
\end{question}

\begin{resolution}{Solução}
  Sendo
  \[
    f(t)= \sqrt{1+t^2}
  \]
  contínua em todo o conjunto dos números reais

  \pause
  Pela parte 1 do Teorema Fundamental do Cálculo
  \[
    g'(x)=\sqrt{1+x^2}
  \]
\end{resolution}
