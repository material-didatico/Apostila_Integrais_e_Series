\begin{frame}{Teorema Fundamental do Cálculo -- Parte 2}
  Se $f$ for {\color{blue} contínua} em $[a, b]$
  \pause
  então, para {\color{blue}qualquer primitiva} $F$ de $f$,
  \pause
  \[
    \int_a^b f(x) \,dx      \pause
    \;=\; F(b) - F(a)       \pause
    \;=\; F(x) \evalat{a}{b}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Demonstração}
  Queremos mostrar que
  \[
    \int_a^b f(x) \,dx = F(b) - F(a)
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Demonstração}
  Pelo Teorema Fundamental do Cálculo -- Parte 1
  \[
    G(x) = \int_a^x f(t)\,dt
  \]
  é uma primitiva de $f$

  \pause
  \vspace{\baselineskip}
  Qualquer outra primitiva tem a forma
  \[
    F(x) = G(x) + C
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema Fundamental do Cálculo -- Parte 2}
  \begin{align*}
    \onslide<+->{F(b) - F(a) & = \big(G(b) + C\big) - \big(G(a) + C\big) \\[5mm]}
    \onslide<+->{            & = G(b) - G(a)                             \\[3mm]}
    \onslide<+->{            & = \int_a^b f(x)\,dx - \int_a^a f(x)\,dx   \\[3mm]}
    \onslide<+->{            & = \int_a^b f(x)\,dx}
  \end{align*}
\end{frame}

\section{Exemplos}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcular a área sob o gráfico de \(f(x)=x^2\) entre 0 e $b$
\end{frame}

\begin{frame}{Exemplo \theexemplo}
  \[
    \text{Área}
    \;=\; \int_0^b f(x) \,dx \pause
    \;=\; \int_0^b x^2  \,dx
  \]
  \pause
  Sabemos que a primitiva de $f$ é
  \[
    F(x)
    \;=\; \int x^2  \,dx \pause
    \;=\; \frac{x^{2+1}}{2+1} + C \pause
    \;=\; \frac{x^3}{3} + C
  \]
  \pause
  então, pelo Teorema Fundamental do Cálculo,
  \[
    \text{Área}
    \;=\; F(x) \evalat{0}{b} \pause
    \;=\; F(b) - F(0)        \pause
    \;=\; \left[ \frac{b^3}{3} + C \right]
    -   \left[ \frac{0^3}{3} + C \right] \pause
    \;=\; \frac{b^3}{3}
  \]
\end{frame}

\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcule \(\int_3^6 \frac{dx}{x}\)
\end{frame}

\begin{frame}{Exemplo \theexemplo}
  \(f(x)=\frac{1}{x}\) é contínua no intervalo $[3, 6]$

  \pause
  \vspace{\baselineskip}
  Uma primitiva de $f$ é $F(x)=\ln\abs{x}$

  \pause
  \vspace{\baselineskip}
  Pelo Teorema Fundamental do Cálculo
  \[
    \int_3^6 \frac{dx}{x}
    \pause
    = \ln\abs{x} \;\evalat{1}{3}
    \pause
    = \ln(6) - \ln(3)
    \pause
    = \ln\left(\frac{6}{3}\right)
    \pause
    = \ln(2)
    \pause
    \approx 0.693
  \]
\end{frame}

\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Calcule
  \(
    \int_0^1 20x^5 - e^x + 3x^6 - \cos(x) + \frac{3}{1+x^2} - 3 \; dx
  \)
\end{frame}

\begin{frame}{Exemplo \theexemplo}
  Queremos uma \emph{integral definida} da função
  \[
    f(x) = 20x^5 - e^x + 3x^6 - \cos(x) + \frac{3}{1+x^2} - 3
  \]

  \pause
  Primeiro precisamos encontrar uma primitiva de $f$

  \pause
  \vspace{0.5\baselineskip}
  Tendo a primitiva, usamos o Teorema Fundamental do Cálculo
\end{frame}

\begin{frame}{Exemplo \theexemplo}
  \begin{align*}
    \onslide<+->{F(x) & = \int f(x) dx \\[2mm]}
    \onslide<+->{& = \int 20x^5 - e^x + 3x^6 - \cos(x) + \frac{3}{1+x^2} - 3\; dx \\[2mm]}
    \onslide<+->{& = \int 20 x^5 dx
      - \int e^x dx
      + \int 3 x^6 dx
      - \int \cos(x) dx
      + \int \frac{3}{1+x^2} dx
      - \int 3 dx \\[2mm]}
    &
      \onslide<+->{= 20\frac{x^6}{6}}
      \onslide<+->{- e^x}
      \onslide<+->{+ 3\frac{x^7}{7}}
      \onslide<+->{- \sin(x)}
      \onslide<+->{+ 3\arctan(x)}
      \onslide<+->{- 3x }
      \onslide<+->{+ C} \\[2mm]
    &
      \onslide<+->{
      = \frac{10}{3}x^6
      - e^x
      + \frac{3}{7}x^7
      - \sin(x)
      + 3\arctan(x)
      - 3x
      + C}
  \end{align*}
\end{frame}
