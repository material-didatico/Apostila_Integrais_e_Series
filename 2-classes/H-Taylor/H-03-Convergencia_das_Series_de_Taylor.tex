\section{Convergência da Série de Taylor}

%------------------------------------------------------------------------------%
\begin{frame}{Polinômios e Séries de Taylor}
  Se $f$ é $k$-derivável, em \(x=a\), temos o \emph{Polinômio de Taylor}
  \[
    P_k(x) = \sum_{n=0}^k \frac{\;f^{(n)\;}(a)}{n!}\,(x-a)^n
  \]

  \pause
  \vspace{\baselineskip}
  Se $f$ é infinitamente derivável, em \(x=a\), temos a \emph{Série de Taylor}
  \[
    S(x) = \sum_{n=0}^\infty \frac{\;f^{(n)\;}(a)}{n!}\,(x-a)^n
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Questões}
  \begin{itemize}[<+->]
    \item Quão próximo o polinômio $\;P_k(x)\;$ está da função $\;f(x)\;$?
    \vspace{\baselineskip}
    \item A série converge, isso é, $\;S(x)\;$ existe?
    \vspace{\baselineskip}
    \item Se $\;S(x)\;$ existe, qual sua relação com $\;f(x)\;$?
  \end{itemize}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Contraexemplo}
  Considere a função
  \[
    f(x) = \begin{cases}
      0                 & x \leq 0 \\[2mm]
      e^{\sfrac{-1}{x}} & x > 0
    \end{cases}
  \]
\end{frame}

\section{Teorema de Taylor}

%------------------------------------------------------------------------------%
\begin{frame}{Teorema de Taylor}
  Sejam
  \begin{itemize}
  \item \(f\colon\mathcal{I}\rightarrow \R\) uma função com \emph{$n+1$
        derivadas contínuas} em $\mathcal{I}$
  \item $a$, $x$ $\in\mathcal{I}$
  \end{itemize}

  \pause
  \vspace{\baselineskip}
  Então existe {\color{red}\(c\)} \emph{entre \(a\) e \(x\)} \pause tal que
  \[
    f(x)=P_n(x)+R_n(x)\
    \qquad \text{onde} \qquad
    R_n(x)=\frac{\;f^{(n+1)}({\color{red}c})\;}{(n+1)!}(x-a)^{n+1}
  \]

  \vspace{.5\baselineskip}
  $R_n(x)\;$ é o \emph{Resto (ou Erro) de Ordem $n$}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Convergência da Série de Taylor}
Se \emph{\(\;R_n(x)\to 0\;\)} quando \emph{\(\;n\to\infty\;\)}
para todo \(\;x\in\mathcal{I}\;\)

\pause
\vspace{\baselineskip}
A Série de Taylor de $f$ converge para $f$ no intervalo \(\;\mathcal{I}\;\)
\[
  f(x) = \sum_{n=0}^\infty \frac{\;f^{(n)\;}(a)}{n!}\,(x-a)^n
\]

\pause
Uma função igual a sua Série de Taylor é chamada \emph{Analítica}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Estimativa de Erro}
Se existir \(\;M\;\) tal que  \emph{\(\;\abs{\,f^{(n+1)}(t)} \leq M \;\)}
para todo \(t\) entre \(x\) e \(a\), inclusive

\pause
\vspace{\baselineskip}
\[
  \abs{R_n(x)}
  = \abs{\frac{\;{\color{blue}f^{(n+1)}({\color{red}c})}\;}{(n+1)!}(x-a)^{n+1}} \pause
  \leq {\color{blue} M}\, \frac{\;\,\abs{x-a}^{n+1}}{(n+1)!\;\;}
\]
\end{frame}

\section{Exemplos}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Estimar o erro ao aproximar \(e^x\), no intervalo \([-1, 1]\),
  por seu Polinômio de Taylor centrado em zero
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Série de Taylor da exponencial
  \[
    f(x) = \sum_{n=0}^\infty \frac{x^n}{n!}
  \]

  \pause
  Erro após a soma de $n$ termos,\(P_n(x)\)
  \[
    R_n(x)=\frac{\;f^{(n+1)}(c)\;}{(n+1)!}x^{n+1}
  \]
  Para algum $c$ entre zero e $x$
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Não temos como saber o valor de $c$\pause, mas sabemos que
  \[
    f^{(n+1)}(c) = e^c \pause \leq e
    \qquad\text{ pois } \quad c \leq 1
  \]
  \pause
  Portanto
  \[
    \abs{R_n(x)} < e\frac{\;\abs{x^{n+1}}\;}{(n+1)!} \pause
                 < \frac{\;e\;}{(n+1)!}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Mostre que a Série de Taylor de \(e^x\) converge para a \(e^x\) para todo
  \(x\in[-1, 1]\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Verificamos que para qualquer \(x\in[-1, 1]\)
  \[
    \abs{R_n(x)} < \frac{\;e\;}{(n+1)!}
    \qquad
    n = 1, 2, 3, \dots
  \]

  \pause
  Como
  \[
    \lim_{n\to\infty} \frac{\;e\;}{(n+1)!} = 0
  \]
  \pause
  O \emph{Teorema do Confronto} garante que
  \(
    \lim_{n\to\infty} R_n(x) = 0
  \)

  \pause
  A Série de Taylor da exponencial converge para a exponencial para \(x\in[-1, 1]\)
\end{frame}

%------------------------------------------------------------------------------%
\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Mostre que a Série de Taylor de \(e^x\) converge para a \(e^x\) para todo
  \(x\in\R\)

  \pause
  \vspace{\baselineskip}
  Sabemos que
  \(
    f^{(n+1)}(c) = e^c
  \)
  para \(c\) entre \(x\) e \(0\)

  \pause
  \vspace{\baselineskip}
  Se \(x>0\), \(c\in[0, x]\)
  \pause
  e \;
  \(
    f^{(n+1)}(c) = e^c \leq e^x
  \)

  \pause
  \vspace{0.5\baselineskip}
  Se \(x<0\), \(c\in[x, 0]\)
  \pause
  e \;
  \(
    f^{(n+1)}(c) = e^c \leq e^0
    \pause
    \leq e^{\abs{x}}
  \)

  \pause
  \vspace{0.5\baselineskip}
  Portanto
  \[
    \abs{f^{(n+1)}(c)} \leq e^{\abs{x}}
    \qquad
    \text{ para todo } x \in \R
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Para qualquer \(x\in\R\)
  \[
    \abs{R_n(x)}
    \;=\;    \frac{\abs{f^{(n+1)}(c)}}{(n+1)!}\abs{x}^{n+1}
    \;\leq\; \frac{\;e^{\abs{x}}\;}{(n+1)!}\abs{x}^{n+1}
  \]

  \pause
  Como
  \[
    \lim_{n\to\infty} \frac{\;e^{\abs{x}}\;}{(n+1)!}\abs{x}^{n+1}
    \;=\; e^{\abs{x}} \abs{x}^{n+1} \lim_{n\to\infty} \frac{1}{(n+1)!}
    \;=\; 0
  \]
  \pause
  o \emph{Teorema do Confronto} garante que
  \(
    \lim_{n\to\infty} R_n(x) = 0
  \)

  \pause
  A Série de Taylor da exponencial converge para a exponencial para \(x\in\R\)
\end{frame}

\addtocounter{exemplo}{1}
\begin{frame}{Exemplo \theexemplo}
  Mostre que a Série de Taylor de \(\cos(x)\) centrada em zero
  converge para a \(\cos(x)\) para todo
  \(x\in\R\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Série de Taylor da função \(\cos(x)\)
  \[
    T(x)
    = \sum_{n=0}^{\infty} \frac{(-1)^n}{(2n)!}\, x^{2n}
    = 1 - \frac{x^2}{2}
    + \frac{x^4}{4!}
    - \frac{x^6}{6!}
    + \cdots
  \]

  \pause
  Erro após a soma de $n$ termos,\(P_n(x)\)
  \[
    R_n(x) = \frac{\;f^{(n+1)}(c)\;}{(n+1)!}x^{n+1}
  \]
  Para algum $c$ entre zero e $x$
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Sabemos que
  \[
    f^{(n+1)}(x)
    \pause
    = \frac{d^{n+1}}{dx^{n+1}} \cos(x)
    \pause
    = \begin{cases}
       \pm \sin(x) & \text{ou} \\
       \pm \cos(x)
    \end{cases}
  \]

  \pause
  \vspace{\baselineskip}
  Portanto
  \[
    \abs{f^{(n+1)}(c)} \leq 1
    \qquad
    \forall n\in\N
    \quad
    \forall c\in\R
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
  Assim
  \[
    \abs{R_n(x)}
    \pause
    \;=\;    \frac{\abs{f^{(n+1)}(c)}}{(n+1)!}\abs{x}^{n+1}
    \pause
    \;\leq\; \frac{\abs{x}^{n+1}}{(n+1)!}
  \]

  \pause
  Como
  \[
    \lim_{n\to\infty} \frac{\abs{x}^{n+1}}{(n+1)!} = 0
  \]
  \pause
  o \emph{Teorema do Confronto} garante que
  \(
    \lim_{n\to\infty} R_n(x) = 0
  \)

  \pause
  A Série de Taylor do cosseno converge para o cosseno para \(x\in\R\)
\end{frame}

\begin{frame}{Exemplo \theexemplo}
  Qual o grau do Polinômio de Taylor necessário para aproximar \(\cos(x)\)

  no intervalo \([-1, 1]\) com erro menor do que \(10^{-6}\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo \theexemplo}
\begin{columns}
  \column{0.5\linewidth}
  Já verificamos que
  \[
    \abs{R_n(x)}
    \;\leq\; \frac{\abs{x}^{n+1}}{(n+1)!}
  \]

  \pause
  \vspace{\baselineskip}
  Como  \(x\in[-1, 1]\)
  \[
    \pause
    \abs{R_n(x)}
    \pause
    \;\leq\; \frac{1^{n+1}}{(n+1)!}
    \pause
    \;=\; \frac{1}{(n+1)!}
  \]

\column{0.5\linewidth}
  \pause
  Precisamos que
  \[
    \frac{1}{(n+1)!} < 10^{-6}
  \]
  \[
    (n+1)! > 10^6
  \]

  \pause
  \vspace{\baselineskip}
  Como \(10! = 3\,628\,800\) temos
  \[
    n = 9
  \]
\end{columns}

\end{frame}
