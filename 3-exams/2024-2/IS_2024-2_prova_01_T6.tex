\exercise{25}
Calcule o valor médio da função
\(
  f(x) = \frac{1 -xe^{-x}}{x}
\)
no intervalo \([1, a]\), para $a>1$

\begin{answer}
  Valor médio
  \[
    \fmedio = \frac{1}{a-1}\int_1^a f(x) dx
  \]
  Calculando a primitiva
  \begin{align*}
    F(x) & = \int f(x) dx \\[2mm]
         & = \int \frac{1 -xe^{-x}}{x} dx \\[2mm]
         & = \int \frac{1}{x} -e^{-x} dx \\[2mm]
         & = \int \frac{1}{x} dx - \int e^{-x} dx \\[2mm]
         & = \ln\abs{x} + e^{-x} + C \\[2mm]
  \end{align*}
  Usando o Teorema Fundamental do Cálculo
  \begin{align*}
    \fmedio & = \frac{1}{a-1}\int_1^a f(x) dx \\[2mm]
            & = \frac{1}{a-1}\left(F(x)\evalat{1}{a}\right) \\[2mm]
            & = \frac{1}{a-1}\big(F(a)-F(1)\big) \\[2mm]
            & = \frac{1}{a-1}\left[
                  \left(\ln\abs{a} + e^{-a} + C\right)
                 -\left(\ln\abs{1} + e^{-1} + C\right)
                \right] \\[2mm]
            & = \frac{1}{a-1}\left[ \ln(a) + e^{-a} -0 - e^{-1} \right] \\[2mm]
            & = \frac{\ln(a) + e^{-a} -e^{-1}}{a-1}
  \end{align*}
\end{answer}

\exercise{25}
Calcule a integral
\(
  \int_0^2 \sqrt{t} - \cos(\pi t) + \sec^2\left(\frac{t}{2}\right) - 3x^3 + 1\; dt
\)

\begin{answer}
  Calculando a primitiva
  \begin{align*}
    F(t) & = \int f(t)\, dt \\[2mm]
         & = \int \sqrt{t}
           - \cos(\pi t)
           + \sec^2\left(\frac{t}{2}\right)
           - 3x^3 + 1 \;dt \\[2mm]
         & = \int t^{\sfrac{1}{2}} \;dt
           - \int \cos(\pi t) \;dt
           + \int \sec^2\left(\frac{t}{2}\right) \;dt
           - \int 3x^3 + 1 \;dt \\[2mm]
         & = \frac{t^{\sfrac{1}{2}+1}}{\sfrac{1}{2}+1}
           - \frac{\sin(\pi t)}{\pi}
           + 2\tan\left(\frac{t}{2}\right)
           - \left(3x^3 + 1\right)t
           + C \\[2mm]
         & = \frac{2}{3}t^{\sfrac{3}{2}}
           - \frac{\sin(\pi t)}{\pi}
           + 2\tan\left(\frac{t}{2}\right)
           - \left(3x^3 + 1\right)t
  \end{align*}
  Calculnado a integral definida
  \begin{align*}
    A & = \int_0^2 f(t)\, dt \\[2mm]
      & = F(t) \evalat{0}{2} \\[2mm]
      & = F(2) - F(0) \\[2mm]
      & = \left[
            \frac{2}{3}2^{\sfrac{3}{2}}
            - \frac{\sin(\pi 2)}{\pi}
            + 2\tan\left(\frac{2}{2}\right)
            - \left(3x^3 + 1\right)2
          \right]
       - \left[
            \frac{2}{3}0^{\sfrac{3}{2}}
            - \frac{\sin(\pi 0)}{\pi}
            + 2\tan\left(\frac{0}{2}\right)
            - \left(3x^3 + 1\right)0
        \right] \\[2mm]
        & = \frac{2^{\sfrac{4}{2}}2^{\sfrac{1}{2}}}{3}
          + 2\tan(1)
          - \left(3x^3 + 1\right)2 \\[2mm]
        & = \frac{4\sqrt{2}}{3} + 2\tan(1) - 6x^3 + 2
  \end{align*}
\end{answer}

\exercise{25}
Considerando as curvas
\[
  y = \cos(x)
  \qquad
  y = \sin(x)
  \qquad
  x = \frac{\pi}{4}
  \qquad\text{e}\qquad
  x = \frac{5\pi}{4}
\]
\begin{enumerate}[label=\alph*)]
  \item Esboce o gráfico das curvas e indique a área delimitada por elas
  \item Calcule a área
\end{enumerate}

\begin{answer}
  \textbf{1)}
  \begin{center}
    \includegraphics[width=0.6\linewidth]{area_entre_curvas_seno}
  \end{center}

  \textbf{2)}

  A função seno passa por cima da cosseno em todo o intervalo, portanto
  \begin{align*}
    A & = \int_{\sfrac{\pi}{4}}^{\sfrac{5\pi}{4}} \sin(x) - \cos(x) dx \\[2mm]
      & = \big(-\cos(x) - \sin(x) \big)\evalat{\sfrac{\pi}{4}}{\sfrac{5\pi}{4}} \\[2mm]
      & = \left(-\cos\left(\frac{5\pi}{4}\right) - \sin\left(\frac{5\pi}{4}\right) \right)
        - \left(-\cos\left(\frac{ \pi}{4}\right) - \sin\left(\frac{ \pi}{4}\right) \right) \\[2mm]
      & = \left(-\frac{-\sqrt{2}}{2} - \frac{-\sqrt{2}}{2} \right)
        - \left(-\frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} \right) \\[2mm]
      & = 4\frac{\sqrt{2}}{2} \\[2mm]
      & = 2\sqrt{2}
  \end{align*}

\end{answer}

\exercise{25}
Encontre o volume do sólido obtido pela rotação
da região limitada pelas curvas
\[
  y = x^3
  \qquad\text{e}\qquad
  y = x^2
\]
em torno da reta
\(
  x = -2
\)

\begin{answer}
  Usando cascas cilíndricas o volume é
  \[
    V = \int_a^b 2\pi r(x) h(x) \,dx
  \]
  As curvas se interceptam nos pontos que satisfazem a equação
  \begin{align*}
    x^2       & = x^3 \\
    x^2 - x^3 & = 0   \\
    x^2(1-x)  & = 0
  \end{align*}
  ou seja, \(x=0\) e \(x=1\)

  O raio é a distância entre o ``$x$ da integração'' e o eixo de rotação \(x=-2\)
  \[
    r(x) = x+2
  \]
  Como \(x^2>x^3\) no intervalo \((0, 1)\) a altura é
  \[
    h(x) = x^2-x^3
  \]
  Temos então
  \begin{align*}
    V & = \int_a^b 2\pi r(x) h(x) \,dx \\[2mm]
      & = \int_0^1 2\pi (x+2) (x^2-x^3) \,dx \\[2mm]
      & = 2\pi \int_0^1 x^3 -x^4 + 2x^2 -2x^3 \,dx \\[2mm]
      & = 2\pi \int_0^1 2x^2 -x^3 -x^4 \,dx \\[2mm]
      & = 2\pi \left(2\frac{x^3}{3} - \frac{x^4}{4} -\frac{x^5}{5}\right)\evalat{0}{1} \\[2mm]
      & = 2\pi \left[
                 \left(2\frac{1^3}{3} -\frac{1^4}{4} - \frac{1^5}{5}\right)
                -\left(2\frac{0^3}{3} -\frac{0^4}{4} - \frac{0^5}{5}\right)
               \right] \\[2mm]
      & = 2\pi \left(\frac{2}{3} -\frac{1}{4} - \frac{1}{5} \right) \\[2mm]
      & = 2\pi \frac{40-15-12}{60} \\[2mm]
      & = \frac{13}{30}\pi
  \end{align*}
\end{answer}
